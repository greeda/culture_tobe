# 위 고정 테이블은 모바일에서만 작동 합니다. (position:sticky;)

# table 내부 라인은 td,th 에서 border 를 그립니다.
# table 외부 라인은 table 에서 border 를 그립니다.
# 고정 테이블은 th의 bottom에 border 그려야 td가 위로 스크롤 될때 border가 사라지지 않습니다.
# 고정 테이블은 th의 right에 border 그려야 td가 왼쪽으로 스크롤 될때 border가 사라지지 않습니다.
# 고정 테이블은 고정되는 부분의 width, height 값을 명시적으로 지정해줘야 합니다.

# 복잡한 테이블 구조에서는 scope 속성을 사용해서 스크린리더가 테이블 내용을 읽기 쉽게 해줍니다.
# thead의 th는 scope="col"
# tbody의 th는 scope="row"
# scope="colgroup", scope="rowgroup" 은 버그가 있어서 사용하지 않습니다.

# 웹접근성 필수 caption 을 작성 합니다.
# caption 작성 방법 - 테이블이름: th,th,th,th.....

# th, td 영역을 텍스트 길이에 상관없이 모두 동일한 크기로 맞추려면 table 태그에 .tbl-cell-equal 클래스를 추가 합니다.
